**For Seekers (Customers, Users, Growth-Seekers):**
- **Profile Creation:** 25 POT (one-time, complete intake form)

- **Referral (Customer):** 250 POT (refer customer who completes engagement)
- **Success Story:** 200 POT (share transformation story with permission to publish)

**For Everyone:**
- **Daily Login:** 1 POT (builds habit, max 30/month)

### Economic Model & Sustainability

**Token Sinks (Removing POT from circulation):**
- All POT spent on premium features is burned (removed from supply)
- Creates deflationary pressure
- Increases scarcity and value over time

**Token Faucets (Adding POT to circulation):**
- Platform mints new POT for contributions
- Controlled emission rate: Max 100,000 POT/month initially
- Adjusted based on marketplace growth

**Inflation Control:**
- Target: 10% annual inflation (net of burned tokens)
- Monitor supply/demand monthly
- Adjust earning rates and burning rates to maintain balance
- Goal: 1 POT ‚âà $1 USD value (soft peg)

**Platform Revenue Streams:**
1. **USD Commission:** 20% on all engagements (unchanged)
2. **Premium Subscriptions:** $50-500/month USD (optional, alternative to POT)
3. **Token Redemption Fee:** 20% when converting POT ‚Üí USD
4. **Enterprise Packages:** Custom pricing for B2B marketplace access

```sql
-- USERS TABLE (Unified identity)
CREATE TABLE users (
    id INTEGER PRIMARY KEY,
    email VARCHAR UNIQUE NOT NULL,
    name VARCHAR NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    pot_balance INTEGER DEFAULT 0,  -- POTENTIAL token balance
    total_pot_earned INTEGER DEFAULT 0,
    total_pot_spent INTEGER DEFAULT 0,
    account_type VARCHAR,  -- 'seeker', 'provider', 'both'
    verified BOOLEAN DEFAULT FALSE,
    rating_average DECIMAL(3,2),  -- 0.00 to 5.00
    rating_count INTEGER DEFAULT 0
);

-- CATEGORIES TABLE (Marketplace categories)
CREATE TABLE categories (
    id INTEGER PRIMARY KEY,
    name VARCHAR NOT NULL,  -- 'financial_advisors', 'career_coaches', etc.
    display_name VARCHAR NOT NULL,  -- 'Financial Advisors', 'Career Coaches'
    description TEXT,
    icon VARCHAR,  -- Icon/emoji for category
    launched_at TIMESTAMP,
    active BOOLEAN DEFAULT TRUE,
    commission_rate DECIMAL(4,2) DEFAULT 20.00,  -- Can vary by category
    matching_criteria JSON  -- Category-specific criteria weights
);

-- SEEKERS TABLE (People looking for growth)
CREATE TABLE seekers (
    id INTEGER PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    category_id INTEGER REFERENCES categories(id),
    needs_description TEXT,
    budget_min INTEGER,
    budget_max INTEGER,
    location VARCHAR,
    format_preference VARCHAR,  -- 'in-person', 'remote', 'either'
    urgency VARCHAR,  -- 'immediate', 'this_week', 'this_month', 'flexible'
    additional_criteria JSON,  -- Category-specific needs
    status VARCHAR DEFAULT 'active',  -- 'active', 'matched', 'engaged', 'completed'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- PROVIDERS TABLE (Service providers across all categories)
CREATE TABLE providers (
    id INTEGER PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    category_id INTEGER REFERENCES categories(id),
    business_name VARCHAR,
    bio TEXT,
    specialties TEXT[],  -- Array of specializations
    pricing_min INTEGER,
    pricing_max INTEGER,
    pricing_model VARCHAR,  -- 'hourly', 'project', 'retainer', 'commission'
    location VARCHAR,
    formats_offered VARCHAR[],  -- ['in-person', 'remote', 'hybrid']
    certifications TEXT[],
    years_experience INTEGER,
    languages VARCHAR[],
    additional_profile JSON,  -- Category-specific profile fields
    status VARCHAR DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    boosted_until TIMESTAMP  -- Profile boost expiration
);

-- MATCHES TABLE (AI-generated matches)
CREATE TABLE matches (
    id INTEGER PRIMARY KEY,
    seeker_id INTEGER REFERENCES seekers(id),
    provider_id INTEGER REFERENCES providers(id),
    category_id INTEGER REFERENCES categories(id),
    match_score DECIMAL(5,2),  -- 0.00 to 100.00
    match_reasoning TEXT,  -- AI-generated explanation
    criteria_breakdown JSON,  -- Scores for each criterion
    status VARCHAR DEFAULT 'suggested',  -- 'suggested', 'viewed', 'contacted', 'engaged', 'declined'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    contacted_at TIMESTAMP,
    engaged_at TIMESTAMP
);

-- ENGAGEMENTS TABLE (Successful matches that converted)
CREATE TABLE engagements (
    id INTEGER PRIMARY KEY,
    match_id INTEGER REFERENCES matches(id),
    seeker_id INTEGER REFERENCES seekers(id),
    provider_id INTEGER REFERENCES providers(id),
    category_id INTEGER REFERENCES categories(id),
    deal_value_usd DECIMAL(10,2),
    commission_usd DECIMAL(10,2),  -- 20% of deal value (or category rate)
    commission_pot INTEGER,  -- 5% of deal value in POT tokens
    status VARCHAR DEFAULT 'confirmed',  -- 'confirmed', 'invoiced', 'paid', 'completed'
    started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP,
    invoice_sent_at TIMESTAMP,
    payment_received_at TIMESTAMP
);

-- RATINGS TABLE (Reviews from both sides)
CREATE TABLE ratings (
    id INTEGER PRIMARY KEY,
    engagement_id INTEGER REFERENCES engagements(id),
    from_user_id INTEGER REFERENCES users(id),
    to_user_id INTEGER REFERENCES users(id),
    rating INTEGER CHECK (rating >= 1 AND rating <= 5),
    review_text TEXT,
    would_recommend BOOLEAN,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    pot_earned INTEGER DEFAULT 10  -- Reward for rating
);

-- POT_TRANSACTIONS TABLE (Internal currency ledger)
CREATE TABLE pot_transactions (
    id INTEGER PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    amount INTEGER,  -- Positive = earned, Negative = spent
    transaction_type VARCHAR,  -- 'earn', 'spend', 'burn', 'redeem'
    category VARCHAR,  -- 'profile_creation', 'match_response', 'engagement_bonus', etc.
    description TEXT,
    reference_type VARCHAR,  -- 'engagement', 'rating', 'referral', etc.
    reference_id INTEGER,  -- ID of related record
    balance_after INTEGER,  -- Running balance
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- REFERRALS TABLE (Track referral bonuses)
CREATE TABLE referrals (
    id INTEGER PRIMARY KEY,
    referrer_id INTEGER REFERENCES users(id),
    referred_id INTEGER REFERENCES users(id),
    referral_type VARCHAR,  -- 'seeker', 'provider'
    status VARCHAR DEFAULT 'pending',  -- 'pending', 'completed', 'rewarded'
    pot_bonus INTEGER,  -- 250 for seeker, 500 for provider
    completed_at TIMESTAMP,
    rewarded_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- CONTENT TABLE (User-generated content for POT rewards)
CREATE TABLE content (
    id INTEGER PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    content_type VARCHAR,  -- 'guide', 'video', 'testimonial', 'success_story'
    title VARCHAR,
    body TEXT,
    url VARCHAR,  -- If external content
    status VARCHAR DEFAULT 'pending',  -- 'pending', 'approved', 'published', 'rejected'
    pot_earned INTEGER,
    approved_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
