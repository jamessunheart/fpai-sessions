    daily_budget=daily_budget

def on_conversion(customer_data):
    # Add to White Rock Ministry
    member = ministry.add_member(
        name=customer_data['name'],
        email=customer_data['email'],
        tier=customer_data['tier'],
        initial_assets=customer_data.get('assets', 0)
    )

    # Log in Sacred Loop
    loop.log_revenue(
        amount=customer_data['paid'],
        service=f"White Rock Ministry - {customer_data['tier'].title()}",
        customer_name=customer_data['name'],
        fulfillment_cost=customer_data.get('fulfillment', 500)
    )

    # Check if we should scale ads
    if loop.get_reinvestment_balance()['available'] > 10000:
        # Scale winners
        for camp in campaign['campaigns']:
            assembly_line.campaign_manager.optimize_budget_allocation(camp['id'])

# Step 5: Daily monitoring (automated)
def daily_check():
    # Get performance
    for camp in campaign['campaigns']:
        analysis = assembly_line.campaign_manager.analyze_split_test(camp['id'])

        if analysis['status'] == 'analyzed':
            print(f"Campaign {camp['name']}")
            print(f"  Winner: {analysis['winner']}")
            print(f"  CPA: ${analysis['winner_cpa']:.2f}")

            # Auto-scale if profitable
            if analysis['winner_cpa'] < 50:  # $50 CPA threshold
                assembly_line.campaign_manager.optimize_budget_allocation(camp['id'])

    # Show loop health
    loop.show_dashboard()
    ministry.show_dashboard()
