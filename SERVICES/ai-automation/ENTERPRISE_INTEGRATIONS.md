"""Apollo.io integration for prospect data"""

import requests
from typing import Dict, List, Optional
import os

class ApolloClient:
    """Apollo.io API client"""

    def __init__(self):
        self.api_key = os.getenv("APOLLO_API_KEY")
        self.base_url = "https://api.apollo.io/v1"

    def search_people(
        self,
        job_titles: List[str],
        company_size: str = "10-100",
        industries: List[str] = None,
        location: str = "United States",
        limit: int = 100
    ) -> List[Dict]:
        """Search for prospects matching ICP"""

        payload = {
            "api_key": self.api_key,
            "q_keywords": None,
            "person_titles": job_titles,
            "organization_num_employees_ranges": [company_size],
            "person_locations": [location],
            "organization_industry_tag_ids": self._get_industry_ids(industries),
            "page": 1,
            "per_page": limit
        }

        response = requests.post(
            f"{self.base_url}/mixed_people/search",
            json=payload
        )

        return response.json()["people"]

    def enrich_person(self, email: str = None, linkedin_url: str = None) -> Dict:
        """Enrich person data"""

        payload = {
            "api_key": self.api_key,
            "email": email,
            "linkedin_url": linkedin_url
        }

        response = requests.post(
            f"{self.base_url}/people/match",
            json=payload
        )

        return response.json()["person"]

    def get_email(self, person_id: str) -> str:
        """Get verified email for person"""

        payload = {
            "api_key": self.api_key,
            "id": person_id
        }

        response = requests.post(
            f"{self.base_url}/emailer_campaigns/email_accounts",
            json=payload
        )

        return response.json()["email"]

"""Instantly.ai integration for cold email"""

import requests
from typing import Dict, List
import os

class InstantlyClient:
    """Instantly.ai API client"""

    def __init__(self):
        self.api_key = os.getenv("INSTANTLY_API_KEY")
        self.base_url = "https://api.instantly.ai/api/v1"

    def create_campaign(
        self,
        name: str,
        email_sequence: List[Dict],
        from_email: str = None
    ) -> str:
        """Create email campaign"""

        payload = {
            "campaign_name": name,
            "sequences": email_sequence,
            "from_email": from_email or self._get_default_email()
        }

        response = requests.post(
            f"{self.base_url}/campaign/create",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json=payload
        )

        return response.json()["campaign_id"]

    def add_leads(
        self,
        campaign_id: str,
        leads: List[Dict]
    ):
        """Add leads to campaign"""

        # Instantly format: email, first_name, last_name, company, etc.
        formatted_leads = [
            {
                "email": lead["email"],
                "first_name": lead.get("first_name"),
                "last_name": lead.get("last_name"),
                "company": lead.get("company_name"),
                "personalization": lead.get("custom_fields", {})
            }
            for lead in leads
        ]

        response = requests.post(
            f"{self.base_url}/campaign/{campaign_id}/add_leads",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={"leads": formatted_leads}
